#include <iostream>

using namespace std;

void order(int numA, int numB, int numC, int *minp, int *maxp, int *medp)
{
    if(numA>=numB){
        *maxp=numA;
        *minp=numB;
    }
    else{
        *maxp=numB;
        *minp=numA;
    }
    if(numC>=*maxp){
        *maxp=numC;
    }
    else if(numC<=*minp){
        *minp=numC;
    }
    *medp=numA+numB+numC-*maxp-*minp;
}

int main()
{
    int numA,numB,numC,min,max,med;

    cin>>numA>>numB>>numC;
    order(numA,numB,numC,&min,&max,&med);

    cout<<min<<endl;
    cout<<med<<endl;
    cout<<max<<endl;


    return 0;
}
